-- TRAZA·OBRA · Firmas de partes (desde el móvil del técnico) create table if not exists public.firma_parte ( fecha date not null, tecnico text not null, firma_inspector text, firma_cliente text, nombre_cliente text, observaciones text, firmado_en timestamptz not null default now(), primary key (fecha, tecnico) ); alter table public.firma_parte enable row level security; drop policy if exists fp_select on public.firma_parte; drop policy if exists fp_insert on public.firma_parte; drop policy if exists fp_update on public.firma_parte; create policy fp_select on public.firma_parte for select to anon, authenticated using (true); create policy fp_insert on public.firma_parte for insert to anon, authenticated with check (true); create policy fp_update on public.firma_parte for update to anon, authenticated using (true) with check (true);
